        MySQL常用知识点

        1. 数据库基础操作

        1.1 数据库管理
        -- 创建数据库
        CREATE DATABASE [IF NOT EXISTS] 数据库名 [CHARACTER SET 字符集];

        -- 示例
        CREATE DATABASE mydb CHARACTER SET utf8mb4;

        -- 查看所有数据库
        SHOW DATABASES;

        -- 使用数据库
        USE 数据库名;

        -- 删除数据库
        DROP DATABASE [IF EXISTS] 数据库名;

        1.2 表操作
        -- 创建表
        CREATE TABLE [IF NOT EXISTS] 表名 (
            列名1 数据类型 [约束条件],
            列名2 数据类型 [约束条件],
            ...
            [表级约束]
        );

        -- 示例：创建学生表
        CREATE TABLE student (
            id INT PRIMARY KEY AUTO_INCREMENT COMMENT '学号',
            name VARCHAR(50) NOT NULL COMMENT '姓名',
            age INT COMMENT '年龄',
            gender CHAR(1) DEFAULT '男' COMMENT '性别',
            email VARCHAR(50) UNIQUE COMMENT '邮箱',
            create_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间'
        );

        -- 查看所有表
        SHOW TABLES;

        -- 查看表结构
        DESC 表名;
        -- 或
        SHOW COLUMNS FROM 表名;

        -- 查看创建表的SQL语句
        SHOW CREATE TABLE 表名;

        -- 修改表名
        ALTER TABLE 旧表名 RENAME TO 新表名;

        -- 添加列
        ALTER TABLE 表名 ADD COLUMN 列名 数据类型 [约束条件] [AFTER 已存在的列名];

        -- 修改列
        ALTER TABLE 表名 MODIFY COLUMN 列名 新数据类型 [新约束条件];

        -- 修改列名
        ALTER TABLE 表名 CHANGE COLUMN 旧列名 新列名 数据类型 [约束条件];

        -- 删除列
        ALTER TABLE 表名 DROP COLUMN 列名;

        -- 删除表
        DROP TABLE [IF EXISTS] 表名;

        2. 数据操作

        2.1 插入数据
        -- 插入完整数据
        INSERT INTO 表名 (列名1, 列名2, ...) VALUES (值1, 值2, ...);

        -- 插入多行数据
        INSERT INTO 表名 (列名1, 列名2, ...) VALUES 
        (第一行值1, 第一行值2, ...),
        (第二行值1, 第二行值2, ...),
        ...;

        -- 示例
        INSERT INTO student (name, age, gender, email) VALUES 
        ('张三', 20, '男', 'zhangsan@example.com'),
        ('李四', 21, '女', 'lisi@example.com');

        2.2 查询数据
        -- 基本查询
        SELECT 列名1, 列名2, ... FROM 表名 [WHERE 条件];

        -- 查询所有列
        SELECT * FROM 表名;

        -- 条件查询
        SELECT * FROM 表名 WHERE 条件;

        -- 排序
        SELECT * FROM 表名 ORDER BY 列名 [ASC|DESC];

        -- 分页查询
        SELECT * FROM 表名 LIMIT 偏移量, 行数;
        -- 或 (MySQL 8.0+)
        SELECT * FROM 表名 LIMIT 行数 OFFSET 偏移量;

        -- 去重
        SELECT DISTINCT 列名 FROM 表名;

        -- 聚合函数
        SELECT COUNT(*) FROM 表名;
        SELECT SUM(列名) FROM 表名;
        SELECT AVG(列名) FROM 表名;
        SELECT MAX(列名) FROM 表名;
        SELECT MIN(列名) FROM 表名;

        -- 分组查询
        SELECT 列名1, 聚合函数(列名2) FROM 表名 GROUP BY 列名1 [HAVING 条件];

        -- 示例：查询每个性别的平均年龄
        SELECT gender, AVG(age) as avg_age FROM student GROUP BY gender;

        注意： WHERE 和 HAVING 的区别：
        - WHERE 在分组前过滤记录，不能包含聚合函数
        - HAVING 在分组后过滤记录，可以包含聚合函数

        2.3 更新数据
        -- 更新数据
        UPDATE 表名 SET 列名1 = 值1, 列名2 = 值2, ... [WHERE 条件];

        -- 示例：更新id为1的学生年龄
        UPDATE student SET age = 22 WHERE id = 1;

        重要： 不带 WHERE 条件的 UPDATE 语句会更新表中所有记录，使用时务必小心！

        2.4 删除数据
        -- 删除数据
        DELETE FROM 表名 [WHERE 条件];

        -- 示例：删除id为2的学生
        DELETE FROM student WHERE id = 2;

        -- 清空表数据（保留表结构）
        TRUNCATE TABLE 表名;

        重要： 不带 WHERE 条件的 DELETE 语句会删除表中所有记录，使用时务必小心！

        3. 多表操作

        3.1 表关系
        MySQL中的表关系主要有三种：
        - 一对一关系：一个表中的一条记录对应另一个表中的一条记录
        - 一对多关系：一个表中的一条记录对应另一个表中的多条记录
        - 多对多关系：两个表中的记录互相对应多条记录，通常需要第三个表作为中间表

        3.2 外键约束
        -- 创建带外键的表
        CREATE TABLE orders (
            order_id INT PRIMARY KEY AUTO_INCREMENT,
            order_date DATETIME DEFAULT CURRENT_TIMESTAMP,
            customer_id INT,
            FOREIGN KEY (customer_id) REFERENCES customers(id)
        );

        -- 添加外键
        ALTER TABLE 表名 ADD CONSTRAINT 约束名 FOREIGN KEY (外键列) REFERENCES 主表(主键列);

        -- 删除外键
        ALTER TABLE 表名 DROP FOREIGN KEY 约束名;

        3.3 连接查询
        -- 内连接（只返回两表中匹配的行）
        SELECT * FROM 表1 INNER JOIN 表2 ON 表1.列名 = 表2.列名;

        -- 左连接（返回左表所有行，右表不匹配则为NULL）
        SELECT * FROM 表1 LEFT JOIN 表2 ON 表1.列名 = 表2.列名;

        -- 右连接（返回右表所有行，左表不匹配则为NULL）
        SELECT * FROM 表1 RIGHT JOIN 表2 ON 表1.列名 = 表2.列名;

        -- 示例：查询订单及其客户信息
        SELECT o.order_id, o.order_date, c.name as customer_name
        FROM orders o
        INNER JOIN customers c ON o.customer_id = c.id;

        3.4 子查询
        -- 子查询作为条件
        SELECT * FROM 表1 WHERE 列名 IN (SELECT 列名 FROM 表2 WHERE 条件);

        -- 子查询作为临时表
        SELECT * FROM (SELECT * FROM 表名 WHERE 条件) AS 临时表名;

        -- 示例：查询价格高于平均价格的商品
        SELECT * FROM products WHERE price > (SELECT AVG(price) FROM products);

        4. 索引

        4.1 索引类型
        - 普通索引：最基本的索引类型，没有唯一性限制
        - 唯一索引：索引列的值必须唯一，但允许有NULL值
        - 主键索引：特殊的唯一索引，不允许有NULL值
        - 复合索引：多个列组合的索引
        - 全文索引：用于全文搜索

        4.2 创建和管理索引
        -- 创建表时指定索引
        CREATE TABLE 表名 (
            列名1 数据类型,
            列名2 数据类型,
            ...,
            INDEX 索引名(列名),
            UNIQUE INDEX 索引名(列名)
        );

        -- 创建索引
        CREATE INDEX 索引名 ON 表名(列名);

        -- 创建唯一索引
        CREATE UNIQUE INDEX 索引名 ON 表名(列名);

        -- 创建复合索引
        CREATE INDEX 索引名 ON 表名(列名1, 列名2, ...);

        -- 查看表的索引
        SHOW INDEX FROM 表名;

        -- 删除索引
        DROP INDEX 索引名 ON 表名;

        索引使用原则：
        - 频繁作为查询条件的字段应创建索引
        - 频繁更新的字段不适合创建索引
        - 索引不是越多越好，会影响插入和更新的性能
        - 复合索引遵循最左前缀原则

        5. 事务

        5.1 事务基本操作
        -- 开始事务
        START TRANSACTION;
        -- 或
        BEGIN;

        -- 提交事务
        COMMIT;

        -- 回滚事务
        ROLLBACK;

        -- 示例
        START TRANSACTION;
        UPDATE accounts SET balance = balance - 100 WHERE id = 1;
        UPDATE accounts SET balance = balance + 100 WHERE id = 2;
        COMMIT;

        5.2 事务特性（ACID）
        - 原子性（Atomicity）：事务是不可分割的工作单位，要么全部执行，要么全部不执行
        - 一致性（Consistency）：事务执行前后，数据库从一个一致性状态变到另一个一致性状态
        - 隔离性（Isolation）：多个事务并发执行时，一个事务的执行不应影响其他事务
        - 持久性（Durability）：事务一旦提交，其结果应永久保存在数据库中

        5.3 事务隔离级别
        隔离级别 | 脏读 | 不可重复读 | 幻读
        --- | --- | --- | ---
        READ UNCOMMITTED（读未提交） | 可能 | 可能 | 可能
        READ COMMITTED（读已提交） | 不可能 | 可能 | 可能
        REPEATABLE READ（可重复读，MySQL默认） | 不可能 | 不可能 | 可能
        SERIALIZABLE（串行化） | 不可能 | 不可能 | 不可能

        -- 查看当前事务隔离级别
        SELECT @@transaction_isolation;

        -- 设置事务隔离级别
        SET [SESSION|GLOBAL] TRANSACTION ISOLATION LEVEL 隔离级别;

        6. 存储过程和函数

        6.1 存储过程
        -- 创建存储过程
        DELIMITER //
        CREATE PROCEDURE 存储过程名(参数列表)
        BEGIN
            -- SQL语句
        END //
        DELIMITER ;

        -- 示例：创建一个添加学生的存储过程
        DELIMITER //
        CREATE PROCEDURE add_student(IN p_name VARCHAR(50), IN p_age INT, IN p_gender CHAR(1))
        BEGIN
            INSERT INTO student(name, age, gender) VALUES(p_name, p_age, p_gender);
        END //
        DELIMITER ;

        -- 调用存储过程
        CALL 存储过程名(参数值列表);

        -- 示例
        CALL add_student('王五', 22, '男');

        -- 查看存储过程
        SHOW PROCEDURE STATUS [LIKE '模式'];

        -- 查看存储过程定义
        SHOW CREATE PROCEDURE 存储过程名;

        -- 删除存储过程
        DROP PROCEDURE [IF EXISTS] 存储过程名;

        6.2 函数
        -- 创建函数
        DELIMITER //
        CREATE FUNCTION 函数名(参数列表) RETURNS 返回类型
        BEGIN
            -- SQL语句
            RETURN 返回值;
        END //
        DELIMITER ;

        -- 示例：创建一个计算学生平均年龄的函数
        DELIMITER //
        CREATE FUNCTION get_avg_age() RETURNS DECIMAL(10,2)
        BEGIN
            DECLARE avg_age DECIMAL(10,2);
            SELECT AVG(age) INTO avg_age FROM student;
            RETURN avg_age;
        END //
        DELIMITER ;

        -- 调用函数
        SELECT 函数名(参数值列表);

        -- 示例
        SELECT get_avg_age();

        -- 查看函数
        SHOW FUNCTION STATUS [LIKE '模式'];

        -- 查看函数定义
        SHOW CREATE FUNCTION 函数名;

        -- 删除函数
        DROP FUNCTION [IF EXISTS] 函数名;

        7. 触发器

        -- 创建触发器
        CREATE TRIGGER 触发器名
        {BEFORE | AFTER} {INSERT | UPDATE | DELETE} ON 表名
        FOR EACH ROW
        BEGIN
            -- 触发器逻辑
        END;

        -- 示例：创建一个在插入学生记录后更新计数表的触发器
        DELIMITER //
        CREATE TRIGGER after_student_insert
        AFTER INSERT ON student
        FOR EACH ROW
        BEGIN
            UPDATE stats SET student_count = student_count + 1;
        END //
        DELIMITER ;

        -- 查看触发器
        SHOW TRIGGERS [FROM 数据库名] [LIKE '模式'];

        -- 删除触发器
        DROP TRIGGER [IF EXISTS] 触发器名;

        8. 视图

        -- 创建视图
        CREATE VIEW 视图名 AS SELECT语句;

        -- 示例：创建一个学生基本信息视图
        CREATE VIEW student_info AS
        SELECT id, name, age, gender FROM student;

        -- 使用视图
        SELECT * FROM 视图名;

        -- 修改视图
        ALTER VIEW 视图名 AS 新SELECT语句;

        -- 删除视图
        DROP VIEW [IF EXISTS] 视图名;

        9. 常见问题与解决方案

        9.1 性能优化
        - 合理使用索引
        - 避免SELECT *，只查询需要的列
        - 使用EXPLAIN分析查询执行计划
        - 合理设计表结构，进行必要的范式化
        - 大表分区或分表
        - 使用连接查询替代子查询
        - 定期优化和维护数据库

        -- 分析查询执行计划
        EXPLAIN SELECT * FROM student WHERE age > 20;

        -- 优化表
        OPTIMIZE TABLE 表名;

        -- 检查表
        CHECK TABLE 表名;

        -- 修复表
        REPAIR TABLE 表名;

        9.2 备份与恢复
        -- 使用mysqldump备份数据库
        mysqldump -u 用户名 -p 数据库名 > 备份文件.sql

        -- 恢复数据库
        mysql -u 用户名 -p 数据库名 < 备份文件.sql
